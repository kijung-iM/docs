1.  Add the whatap-node-agent sidecar container to collect the user Pod resource usage.

    ```ini
    ...
      - name: whatap-node-agent
        image: whatap/kube_mon_sidecar
        resources:
          requests:
            memory: 20Mi
            cpu: 10m
          limits:
            memory: 40Mi
            cpu: 20m
        ports:
          - name: nodeport
            containerPort: 6600
        env:
          - name: NODE_IP
            valueFrom:
              fieldRef:
                fieldPath: status.hostIP
          - name: NODE_NAME
            valueFrom:
              fieldRef:
                fieldPath: spec.nodeName
            envFrom:
            - configMapRef:
                name: node-license
    volumes:
    ...
    ```
